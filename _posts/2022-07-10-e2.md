### [문자열 - 예시 코드]

```typescript
// string
let car:string = "HYUNDAI";
console.log(car);
car = "KIA"
console.log(car);
```

```
HYUNDAI
KIA
```



### [정수형 - 예시코드]

```typescript
// int
let age: number = 35;
console.log(age);
```

```
35
```



### [bool타입 - 예시 코드]

```typescript
// boolean

```

```

```



### [배열 - 예시 코드]

```typescript
// array
let a1: number[] = [1, 2, 3];
let a2: Array<string> = ['four', 'five', 'six'];
console.log(a1, a2);
```

```
[ 1, 2, 3 ] [ 'four', 'five', 'six' ]
```





### [튜플 - 예시 코드]

```typescript
// 튜플 (Tuple)
let b: [string, number] = ['abc', 1];
console.log(b[0]);
console.log(b[1]);
```

```
abc
1
```

### [enum - 예시 코드]

```typescript
// enum 열거형
enum Person {
    Name = "LeeDongKyoo",
    Age = 35,
    Job = "Programmer"
}
console.log(Person.Name, Person.Age, Person.Job);
console.log(Person["LeeDongKyoo"], Person[35], Person["Programmer"]);

// enum 열거형 전달
let Person1: Person;
Person1 = Person.Name;
console.log(Person1);
```

```
LeeDongKyoo 35 Programmer
undefined Age undefined
LeeDongKyoo
```
